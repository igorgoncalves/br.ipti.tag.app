            (state as ClassroomUpdateDeleteFormState).weekDaysWednesday,
      );
      _usecaseUpdateClassroom.call(
        Params(
          event.id,
          params,
        ),
      );
    } else if (event is DeleteClassroom) {
      _usecaseDeleteClassroom.call(
        event.id,
      );
    }

    yield newState;
  }
